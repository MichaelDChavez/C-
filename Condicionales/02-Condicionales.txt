using System;

class Program
{

    {
        Console.Write("Ingrese su nombre: ");
        string nombre = Console.ReadLine();
        
        Console.Write("Ingrese su edad: ");
        int edad = int.Parse(Console.ReadLine());

        if (edad < 0 || edad > 100)
        {
            Console.WriteLine("Ingrese una edad valida.");
        }
