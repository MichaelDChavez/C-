using System;
class Program
{
    static void Main()
    {
        // Número de filas del triángulo
        int filas = 6;
        // Generar el triángulo
        for (int i = 1; i <= filas; i++)
        {
            // Imprimir números en cada fila
            for (int j = 1; j <= i; j++)
            {
                Console.Write(j);
            }
            Console.WriteLine(); // Salto de línea después de cada fila
        }
    }
}
